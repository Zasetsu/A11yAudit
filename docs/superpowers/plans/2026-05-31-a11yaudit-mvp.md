# A11yAudit MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first working A11yAudit MVP: a TypeScript monorepo with a shared audit engine, CLI, self-hosted API, React Web UI based on the reference prototype, local storage, SQLite persistence, and PDF report generation.

**Architecture:** The system is a local-first TypeScript monorepo. `packages/*` contain the reusable audit engine, crawler, rules, reporter, and storage logic. `apps/cli` runs audits directly, `apps/server` orchestrates scans and persists results, and `apps/web` provides the React UI using the approved prototype as the visual baseline.

**Tech Stack:** pnpm workspaces, TypeScript, Vite, React, Fastify, Drizzle ORM, SQLite, Vitest, Playwright, axe-core, commander, local filesystem storage.

Create `packages/storage/src/local-storage.ts`:

```ts
import { mkdir, readFile, rm, writeFile } from "node:fs/promises";
import { dirname, join, normalize } from "node:path";
import type { StorageAdapter, StoredArtifact } from "./storage";

export interface LocalStorageOptions {
  rootDir: string;
}

export class LocalStorageAdapter implements StorageAdapter {
  constructor(private readonly options: LocalStorageOptions) {}

  async put(key: string, body: Buffer, mimeType: string): Promise<StoredArtifact> {
    const path = this.resolveKey(key);
    await mkdir(dirname(path), { recursive: true });
    await writeFile(path, body);
    return { key, mimeType, sizeBytes: body.byteLength };
  }

  async get(key: string): Promise<Buffer> {
    return readFile(this.resolveKey(key));
  }

  async delete(key: string): Promise<void> {
    await rm(this.resolveKey(key), { force: true });
  }

  private resolveKey(key: string): string {
    const normalized = normalize(key).replace(/^(\.\.(\/|\\|$))+/, "");
    return join(this.options.rootDir, normalized);
  }
}

const TRACKING_PARAMS = ["fbclid", "gclid", "yclid"];
const STATIC_EXTENSIONS = [".pdf", ".zip", ".jpg", ".jpeg", ".png", ".gif", ".webp", ".mp4", ".docx", ".xlsx"];

export function normalizeAuditUrl(input: string): string {
  const url = new URL(input);
  url.hash = "";

  for (const key of [...url.searchParams.keys()]) {
    if (key.startsWith("utm_") || TRACKING_PARAMS.includes(key)) {
      url.searchParams.delete(key);
    }
  }

  const pathname = url.pathname !== "/" && url.pathname.endsWith("/")
    ? url.pathname.slice(0, -1)
    : url.pathname;
  url.pathname = pathname;

  const query = url.searchParams.toString();
  return `${url.origin}${url.pathname}${query ? `?${query}` : ""}`;
}

export function shouldSkipUrl(input: string): boolean {
  try {
    const url = new URL(input);
    if (!["http:", "https:"].includes(url.protocol)) return true;
    return STATIC_EXTENSIONS.some((ext) => url.pathname.toLowerCase().endsWith(ext));
  } catch {
    return true;
  }
}
```

Create `packages/crawler/src/network-safety.ts`:

```ts
import { isIP } from "node:net";

export function isBlockedHostnameOrIp(hostname: string): boolean {
  const host = hostname.toLowerCase();
  if (host === "localhost" || host.endsWith(".localhost")) return true;

  if (isIP(host) === 4) {
    const parts = host.split(".").map(Number);
    const [a, b] = parts;
    if (a === 10) return true;
    if (a === 127) return true;
    if (a === 169 && b === 254) return true;
    if (a === 172 && b >= 16 && b <= 31) return true;
    if (a === 192 && b === 168) return true;
  }

  if (isIP(host) === 6) {
    if (host === "::1") return true;
    if (host.startsWith("fe80:")) return true;
    if (host.startsWith("fc") || host.startsWith("fd")) return true;
  }

  return false;
}

export function assertSafeUrl(input: string): void {
  const url = new URL(input);
  if (isBlockedHostnameOrIp(url.hostname)) {
    throw new Error(`Blocked unsafe audit target: ${url.hostname}`);
  }
}
```

import { DEFAULT_SCAN_LIMITS } from "@a11yaudit/core";
import { assertSafeUrl } from "./network-safety";
import { normalizeAuditUrl, shouldSkipUrl } from "./url-normalizer";

export interface CrawlInput {
  startUrl: string;
  maxPages?: number;
}

export interface CrawlResult {
  urls: string[];
}

export async function crawlStaticSeed(input: CrawlInput): Promise<CrawlResult> {
  assertSafeUrl(input.startUrl);
  if (shouldSkipUrl(input.startUrl)) return { urls: [] };
  return { urls: [normalizeAuditUrl(input.startUrl)].slice(0, input.maxPages ?? DEFAULT_SCAN_LIMITS.maxPages) };
}

export function wcagTagsToCriteria(tags: string[]): string[] {
  const criteria = new Set<string>();
  for (const tag of tags) {
    const match = /^wcag(\d)(\d)(\d)$/.exec(tag);
    if (match) criteria.add(`${match[1]}.${match[2]}.${match[3]}`);
  }
  return [...criteria].sort();
}

import { mkdir } from "node:fs/promises";

import { program } from "commander";

  .name("a11y-audit")
  .description("Self-hosted WCAG 2.2 technical accessibility audit tool")
  .version("0.1.0");

program.command("scan")
  .argument("<url>", "public URL to audit")
  .option("--pdf", "generate a PDF report")
  .option("--out <dir>", "artifact output directory", ".a11yaudit")
  .action(async (url: string, options: { pdf?: boolean; out: string }) => {

    const storage = new LocalStorageAdapter({ rootDir: options.out });
    const crawl = await crawlStaticSeed({ startUrl: url });

      score: 100,
      pagesAudited: crawl.urls.length,
      findingsTotal: 0,
      generatedAt: new Date().toISOString(),
      findings: []
    });

    await storage.put("latest-report.html", Buffer.from(html), "text/html");

    if (options.pdf) {
      const pdf = await renderPdfFromHtml(html);
      await storage.put("latest-report.pdf", pdf, "application/pdf");
    }

    console.log(`A11yAudit completed: ${crawl.urls.length} page(s) processed`);
    console.log(`Artifacts written to ${join(process.cwd(), options.out)}`);
  });

await program.parseAsync();

export type LocalJobStatus = "queued" | "running" | "completed" | "failed";

export interface LocalJob<TPayload> {
  id: string;
  status: LocalJobStatus;
  payload: TPayload;
  error?: string;
}

export class LocalJobRunner<TPayload> {
  private readonly jobs = new Map<string, LocalJob<TPayload>>();

  enqueue(id: string, payload: TPayload): LocalJob<TPayload> {
    const job: LocalJob<TPayload> = { id, payload, status: "queued" };
    this.jobs.set(id, job);
    return job;
  }

  get(id: string): LocalJob<TPayload> | undefined {
    return this.jobs.get(id);
  }

  list(): LocalJob<TPayload>[] {
    return [...this.jobs.values()];
  }
}
